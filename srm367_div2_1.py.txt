class WhiteCells(object):
    def countOccupied(self,board):
        self.board=""
        count=0
        board=list(board)
        for i in range(len(board)):
            c_1=board[i]
            if i%2==0:
                for j in range(0,len(c_1),2):
                    if c_1[j]=='F':
                        count+=1
            else:
                for j in range(1,len(c_1),2):
                    if c_1[j]=='F':
                        count+=1
        return count
